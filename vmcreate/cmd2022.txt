### --- 4. iSCSIターゲットの作成と割り当て --- ###

# iSCSIターゲット名
$targetName = "Target01"

# 接続を許可するクライアントのIPアドレスリスト
$initiatorIPs = @(
    "10.10.22.11",
    "10.10.22.12"
)

# IPアドレス形式の Initiator ID に変換

#■■■■■■RDS構成
# --- 変数定義 ---
$ConnectionBroker = "win2022RDSsb.contosoRDS.local"
$WebAccessServer = "win2022RDSsb.contosoRDS.local"
$LicenseServer = "win2022RDSsb.contosoRDS.local"
$SessionHost = "win2022RDSSH.contosoRDS.local"
$CollectionName = "AppCollection"
$CollectionDesc = "Test collection with notepad"
$UserGroup = "Domain Users"  # アクセス許可を与えるグループ

# --- 1. RDS関連の役割をインストール ---
$servers = @($ConnectionBroker, $SessionHost)
foreach ($srv in $servers) {
    Invoke-Command -ComputerName $srv -ScriptBlock {
        Install-WindowsFeature -Name RDS-RD-Server -IncludeManagementTools -Restart:$false
    }
}
# ブローカー／ライセンス／Webアクセスの役割をまとめて追加
Invoke-Command -ComputerName $ConnectionBroker -ScriptBlock {
    Install-WindowsFeature -Name RDS-Connection-Broker, RDS-Web-Access, RDS-Licensing -IncludeManagementTools -Restart:$false
}

#★再起動必要。セッションホストを管理サーバーへ追加が必要、

# --- 2. RDS標準デプロイを作成 ---
New-RDSessionDeployment `
    -ConnectionBroker $ConnectionBroker `
    -SessionHost $SessionHost `
    -WebAccessServer $WebAccessServer

# --- 3. ライセンスサーバーを追加 + ライセンスモード設定（PerUser） ---
Add-RDServer `
    -Server $LicenseServer `
    -Role RDS-Licensing `
    -ConnectionBroker $ConnectionBroker

Set-RDLicenseConfiguration `
    -LicenseServer $LicenseServer `
    -Mode PerUser `
    -ConnectionBroker $ConnectionBroker

# --- 4. セッションコレクション作成 ---
New-RDSessionCollection `
    -CollectionName $CollectionName `
    -SessionHost $SessionHost `
    -ConnectionBroker $ConnectionBroker `
    -CollectionDescription $CollectionDesc

# --- 5. 公開アプリ（メモ帳）を追加 ---
New-RDRemoteApp `
    -Alias "Notepad" `
    -DisplayName "メモ帳" `
    -FilePath "C:\Windows\System32\notepad.exe" `
    -CollectionName $CollectionName `
    -ConnectionBroker $ConnectionBroker

# --- 6. Webアクセス許可ユーザーを設定 ---
Set-RDWebAccessUser `
    -UserGroup $UserGroup `
    -CollectionName $CollectionName `
    -ConnectionBroker $ConnectionBroker

# --- 7. 結果確認 ---
Write-Host "`n===== 構成完了！ =====`n"
Get-RDDeployment
Get-RDSessionCollection -ConnectionBroker $ConnectionBroker
Get-RDRemoteApp -CollectionName $CollectionName -ConnectionBroker $ConnectionBroker
