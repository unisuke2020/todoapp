# ToDoアプリ 実装計画

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** スマホ対応のToDoリストWebアプリを `todo/index.html` に作成し、Netlifyで自動公開する。

**Architecture:** ビルドステップなしの単一HTMLファイル（HTML + CSS + JS インライン）。データは `localStorage` に JSON 配列で永続化。Netlifyは `netlify.toml` で `todo/` ディレクトリを公開ディレクトリとして指定。

**Tech Stack:** バニラHTML/CSS/JS、localStorage、Netlify無料プラン

### Task 2: JSデータ層とrender関数を実装

**Files:**
- Modify: `todo/index.html` の `<script>` タグ内

- [ ] **Step 1: `<script>` タグの中身をすべて以下に置き換える**

```javascript
const STORAGE_KEY = 'todo-items';

function load() {
  try { return JSON.parse(localStorage.getItem(STORAGE_KEY)) || []; }
  catch { return []; }
}

function save(items) {
  localStorage.setItem(STORAGE_KEY, JSON.stringify(items));
}

function escapeHtml(text) {
  return text
    .replace(/&/g, '&amp;')
    .replace(/</g, '&lt;')
    .replace(/>/g, '&gt;')
    .replace(/"/g, '&quot;');
}

let items = load();
let filter = 'all';

function filtered() {
  if (filter === 'active') return items.filter(i => !i.done);
  if (filter === 'done')   return items.filter(i => i.done);
  return items;
}

function render() {
  const list = document.getElementById('todo-list');
  const visible = filtered();
  if (visible.length === 0) {
    list.innerHTML = '<li class="empty">タスクはありません</li>';
    return;
  }
  list.innerHTML = visible.map(item => `
    <li class="todo-item${item.done ? ' done' : ''}" data-id="${item.id}">
      <input type="checkbox" ${item.done ? 'checked' : ''} onchange="toggle(${item.id})">
      <span>${escapeHtml(item.text)}</span>
      <button class="delete-btn" onclick="remove(${item.id})">🗑</button>
    </li>
  `).join('');
}

render();
```

### Task 3: タスク追加・完了・削除・フィルター機能を実装

**Files:**
- Modify: `todo/index.html` の `<script>` タグ末尾に追加

- [ ] **Step 1: `render();` の直後に以下を追記する**

```javascript
function addItem(text) {
  text = text.trim();
  if (!text) return;
  items.push({ id: Date.now(), text, done: false });
  save(items);
  render();
}

function toggle(id) {
  const item = items.find(i => i.id === id);
  if (item) { item.done = !item.done; save(items); render(); }
}

function remove(id) {
  items = items.filter(i => i.id !== id);
  save(items);
  render();
}

function setFilter(f) {
  filter = f;
  document.querySelectorAll('.filter-btn').forEach(btn => {
    btn.classList.toggle('active', btn.dataset.filter === f);
  });
  render();
}

// イベントリスナー
document.getElementById('add-btn').addEventListener('click', () => {
  const input = document.getElementById('new-todo');
  addItem(input.value);
  input.value = '';
  input.focus();
});

document.getElementById('new-todo').addEventListener('keydown', e => {
  if (e.key === 'Enter') {
    addItem(e.target.value);
    e.target.value = '';
  }
});

document.querySelectorAll('.filter-btn').forEach(btn => {
  btn.addEventListener('click', () => setFilter(btn.dataset.filter));
});
```
